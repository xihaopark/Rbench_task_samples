#!/usr/bin/env Rscript

suppressPackageStartupMessages({
  library(data.table)
})

# Create outputs directory if it doesn't exist
if (!dir.exists("outputs")) dir.create("outputs", recursive = TRUE)

# Read input files
conc_col  <- fread("inputs/conc_col.tsv")$conc_col[1]
data_dt   <- fread("inputs/data.tsv")
groups_dt <- fread("inputs/groups.tsv")$groups
matrix_col <- fread("inputs/matrix_col.tsv")$matrix_col[1]
spec1     <- fread("inputs/spec1.tsv")$spec1[1]
spec2     <- fread("inputs/spec2.tsv")$spec2[1]
units_col <- fread("inputs/units_col.tsv")$units_col[1]

# Ensure required columns exist
req_cols <- c("USUBJID", matrix_col, conc_col, units_col, "ARM")
missing_cols <- setdiff(req_cols, names(data_dt))
if (length(missing_cols) > 0) {
  stop("Missing required columns in data.tsv: ", paste(missing_cols, collapse = ", "))
}

setDT(data_dt)

# Subset for the two specified matrices
d1 <- data_dt[get(matrix_col) == spec1]
d2 <- data_dt[get(matrix_col) == spec2]

# Merge on grouping variables (from groups.tsv)
by_vars <- groups_dt
if (!all(by_vars %in% names(data_dt))) {
  stop("Grouping variables from groups.tsv not all present in data.tsv")
}

setkeyv(d1, by_vars)
setkeyv(d2, by_vars)

merged_dt <- merge(
  d1[, c(by_vars, conc_col, units_col, matrix_col), with = FALSE],
  d2[, c(by_vars, conc_col, units_col, matrix_col), with = FALSE],
  by = by_vars,
  suffixes = c("_1", "_2")
)

# Check unit consistency
merged_dt <- merged_dt[get(paste0(units_col, "_1")) == get(paste0(units_col, "_2"))]

# Compute ratio: spec1 / spec2
merged_dt[, CONC := get(paste0(conc_col, "_1")) / get(paste0(conc_col, "_2"))]

# Use units from first matrix (assumed same)
merged_dt[, CONCU := get(paste0(units_col, "_1"))]

# Set MATRIX to ratio label
merged_dt[, MATRIX := paste0(spec1, "/", spec2)]

# Add task identifier
merged_dt[, rbio_reference_task := "aNCA_clinical_multiple_matrix_ratio"]

# Prepare final result with required columns
result <- merged_dt[, .(
  USUBJID,
  MATRIX,
  CONC,
  CONCU,
  ARM,
  rbio_reference_task
)]

# Write output
fwrite(result, file = "outputs/result.csv")
