## Reference Code

```r
#!/usr/bin/env Rscript
suppressPackageStartupMessages(library(GSEABase))

gmt_path <- file.path('inputs', 'signature.gmt')
if (!file.exists(gmt_path)) stop('signature.gmt input is missing.')

gsc <- getGmt(gmt_path)
if (length(gsc) == 0) stop('GeneSetCollection is empty.')

inc_mat <- incidence(gsc)
inc_mat <- as.matrix(inc_mat)

overlap_mat <- t(inc_mat) %*% inc_mat
overlap_df <- as.data.frame(overlap_mat)
overlap_df <- cbind(set_name = rownames(overlap_mat), overlap_df)

set_names <- rownames(overlap_mat)
summary_list <- lapply(seq_along(set_names), function(i) {
  row_vals <- overlap_mat[i, ]
  row_vals[i] <- NA
  partner_idx <- which.max(row_vals)
  if (length(partner_idx) == 0 || is.na(row_vals[partner_idx])) {
    partner_name <- NA_character_
    overlap_size <- 0
  } else {
    partner_name <- set_names[partner_idx]
    overlap_size <- as.numeric(row_vals[partner_idx])
  }
  data.frame(
    set_name = set_names[i],
    best_partner = partner_name,
    overlap_size = overlap_size,
    self_size = diag(overlap_mat)[i],
    stringsAsFactors = FALSE
  )
})

overlap_summary <- do.call(rbind, summary_list)

dir.create('outputs', showWarnings = FALSE, recursive = TRUE)
write.csv(overlap_df, file.path('outputs', 'overlap_matrix.csv'), row.names = FALSE)
write.csv(overlap_summary, file.path('outputs', 'overlap_summary.csv'), row.names = FALSE)
```

```r
# Load required package
library(GSEABase)

# Create outputs directory if it doesn't exist
dir.create('outputs', showWarnings = FALSE)

# Read GMT file
gmt_path <- 'inputs/signature.gmt'
if (!file.exists(gmt_path)) {
    stop("Input file not found: ", gmt_path)
}

gene_set_collection <- getGmt(gmt_path)

# Convert GeneSetCollection to incidence matrix (gene sets × genes)
inc_mat <- incidence(gene_set_collection)

# Calculate pairwise overlap counts: overlap = inc_mat %*% t(inc_mat)
# This gives a square matrix where entry (i,j) is number of genes shared between set i and set j
overlap_counts <- inc_mat %*% t(inc_mat)

# Convert to standard R matrix (from class 'dgCMatrix' if sparse)
overlap_matrix <- as.matrix(overlap_counts)

# Write full overlap matrix
write.csv(overlap_matrix, file = 'outputs/overlap_matrix.csv', quote = FALSE)

# Create summary data frame
set_names <- rownames(overlap_matrix)
n_sets <- length(set_names)

# Initialize summary vectors
set_size <- diag(overlap_matrix)          # Number of genes in each set (self-overlap)
max_partner <- character(n_sets)          # Partner with largest non‑self overlap
max_overlap <- numeric(n_sets)            # Size of that overlap

for (i in 1:n_sets) {
    # Get overlaps for set i, exclude self
    overlaps <- overlap_matrix[i, ]
    overlaps[i] <- 0
    
    if (any(overlaps > 0)) {
        idx <- which.max(overlaps)
        max_partner[i] <- set_names[idx]
        max_overlap[i] <- overlaps[idx]
    } else {
        max_partner[i] <- NA
        max_overlap[i] <- 0
    }
}

summary_df <- data.frame(
    gene_set = set_names,
    set_size = set_size,
    max_partner = max_partner,
    max_overlap = max_overlap
)

# Write summary
write.csv(summary_df, file = 'outputs/overlap_summary.csv', quote = FALSE, row.names = FALSE)

# Basic quality checks
cat("Output files created:\n")
cat("  - outputs/overlap_matrix.csv\n")
cat("  - outputs/overlap_summary.csv\n\n")

cat("First few rows of overlap_summary.csv:\n")
print(head(summary_df))

cat("\nChecking for anomalies...\n")
# 1. No NA in overlap matrix (except possibly for missing genes, but matrix should be complete)
if (any(is.na(overlap_matrix))) {
    warning("Overlap matrix contains NA values.")
} else {
    cat("✓ Overlap matrix has no NA values.\n")
}

# 2. Set sizes (diagonal) should be positive integers
if (any(set_size <= 0)) {
    warning("Some gene sets have non‑positive size.")
} else {
    cat("✓ All gene‑set sizes are positive.\n")
}

# 3. Overlap matrix should be symmetric (within numeric tolerance)
if (!isSymmetric(overlap_matrix, tol = 1e-10)) {
    warning("Overlap matrix is not symmetric (beyond tolerance).")
} else {
    cat("✓ Overlap matrix is symmetric.\n")
}

# 4. Self‑overlap equals set size (diagonal)
if (!all(diag(overlap_matrix) == set_size)) {
    warning("Diagonal of overlap matrix does not match computed set sizes.")
} else {
    cat("✓ Diagonal entries match set sizes.\n")
}

cat("\nDone.\n")
```
